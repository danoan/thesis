\subsection*{Elastica energy and the completion property}

\begin{frame}
{The elastica energy}

Let $X$ an Euclidean shape and $\alpha >0, \beta>0$.

\begin{align*}
 E(X) = \int_{\partial X}{ \alpha + \beta \kappa ^2ds}. 
\end{align*}


\end{frame}
